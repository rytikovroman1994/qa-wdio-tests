  // Domain Page
  browser.waitForVisible('.wizard-main-body.ul-wizard');

  browser.waitForExist('.w-domain-tmp-change.js-domain-tmp-change');
  browser.waitForExist('.w-next-step > button.ul-button-link.js-prev-btn');
  browser.waitForExist('.w-next-step > button.ul-button.js-next-btn');

  browser.click('.w-domain-editable.w-domain-name.text.js-w-domain-editable', (response) => {
    this.assert.ok(browser === this, 'Check if the context is right.');
    this.assert.ok(typeof response === 'object', 'We got a response object.');
    browser.pause(2000);
  });

  browser.waitForExist('select > option');
}
